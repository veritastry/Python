def  get_search_field(self,request):
        print('++++++'*1000)
        from   django.db.models  import Q
        get_search_content=request.GET.get('search_field')
        q=Q()
        q.connector='or'
        if  get_search_content:
            for  search_field  in self.search_fields:
                q.children.append((search_field+'__contains',get_search_content))
        return q
